# Update project

Updates a project and with new data from the request body.

# API information
URL localhost:3600/project/:id
Method POST
Request body 
{
  "name": "My new project name"
}
Request headers Authorization: Bearer <access-token>
Response body 
{
  "children":[],
  "params":["EDP$DefaultParameters"],
  "_id":"60a860949d27de857423a67f",
  "name":"My new project name",
  "owner":"60907b23eddd2e1ef4270f9b",
  "lastAccessed":"2021-05-22T01:38:28.828Z"
}
Response status code 200

# Request body fields

name: string

A new name for the project.

Note: The API will allow other fields in the request, however, only the name field will be used for updating the project.

# Response body

The response body contains a Project object containing the result of the update.

# Example usage

    ///////////////////////////////////////////////////////////////////////////////
// The following code invokes the Logout API to terminate a session and      //
// invalidate the current access token                                       //
///////////////////////////////////////////////////////////////////////////////
const axios = require('axios');

const projectId = 'some project ID here';
const data = {
    "name": "My new project name"
}

axios.post(
  'http://localhost:3600/auth',
  {
    email: 'john.doe@org.com',
    password: 'password',
  },
)
  .then((res) => {
    // Authentication successful
    const { accessToken } = res.data;

    // updating the project
    axios.post(
      `http://localhost:3600/project/${projectId}`,
      data,
      {
        headers: {
          Authorization: `Bearer ${accessToken}`,
        },
      },
    )
      // on success output the returned data
      .then((res) => console.log(res.data))
      // on error report failure
      .catch(() => console.log('Failed to update the project'));
  })
  .catch(() => console.log('Authentication failed'));

    ###############################################################################
# The following code invokes the Logout API to terminate a session and        #
# invalidate the current access token                                         #
###############################################################################

# importing the requests library
import requests

projectId = 'some project ID here'

# setup the data to be passed to the API
data = {'email': 'john.doe@org.com',
        'password': 'password'
}

# sending login request and saving response as response object
r = requests.post(url = 'http://localhost/3600/auth', data = data)

# get the login response
loginResponse = r.json()

# opening the project
data = {
    "name": "My new project name"
}
headers  = { 'Authorization': f'Bearer {loginResponse.accessToken}' }
r = requests.post(
  url = f'http://localhost/3600/project/{projectId}',
  data = data,
  headers = headers
)

# print the response data
if r.status_code == 200
  print(r.json())
else
  print('Failed to update the project')
