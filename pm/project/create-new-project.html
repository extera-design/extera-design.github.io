# Create new project

Creates a new project for an authenticated user.

# API information
URL localhost:3600/project
Method POST
Request body 
{
    "name": "My project",
    "params": ["EDP$DefaultParameters"]
}
Request headers Authorization: Bearer <access-token>
Response body 
{
  "_id":"60a860949d27de857423a67f",
  "children":[],
  "params":["EDP$DefaultParameters"],
  "name":"My project",
  "owner":"60907b23eddd2e1ef4270f9b",
  "lastAccessed":"2021-05-22T01:38:28.762Z"
}
Response status code 201

# Request body fields

name: string

The name for this project

params: Array<string>

The list of IDs for the parameter groups supported by this project. All cases created directly in this project will use all the parameters in the project's groups. However, studies may choose a subset of the project's parameter groups and cases created under the study will inherit the study's parameters. If the list contains IDs that do not represent parameter groups, they are removed from the list retained for the project.

# Response body

Contains a Project object providing the data for the newly created project.

# Example usage

    ///////////////////////////////////////////////////////////////////////////////
// The following code invokes the Logout API to terminate a session and      //
// invalidate the current access token                                       //
///////////////////////////////////////////////////////////////////////////////
const axios = require('axios');

axios.post(
  'http://localhost:3600/auth',
  {
    email: 'john.doe@org.com',
    password: 'password',
  },
)
  .then((res) => {
    // Authentication successful
    const { accessToken } = res.data;

    // creating a new project
    axios.post(
      'http://localhost:3600/project',
      {
        name: 'Name for new project',
        params: ['EDP$DefaultParameters'],
      },
      {
        headers: {
          Authorization: `Bearer ${accessToken}`,
        },
      },
    )
      // on success output the returned data
      .then((res) => console.log(res.data))
      // on error report failure
      .catch(() => console.log('Failed to create the new project'));
  })
  .catch(() => console.log('Authentication failed'));

# The following code invokes the Logout API to terminate a session and        #
# invalidate the current access token                                         #
###############################################################################

# importing the requests library
import requests

# setup the data to be passed to the API
data = {'email': 'john.doe@org.com',
        'password': 'password'
}

# sending login request and saving response as response object
r = requests.post(url = 'http://localhost/3600/auth', data = data)

# get the login response
loginResponse = r.json()

# creating the project
headers  = { 'Authorization': f'Bearer {loginResponse.accessToken}' }
data = { 'name': 'Name for new project', 'params': [ 'EDP$DefaultParameters' ] }
r = requests.post(url = 'http://localhost/3600/project', data = data, headers = headers)

# print the response data
if r.status_code == 201
  print(r.json())
else
  print('Failed to create the new project')
