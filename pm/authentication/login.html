# Login

Logins a user by validating the user e-mail and password and, if validation succeeds, returning an access token that will provide access to the rest of the APIs.

# API information
URL localhost:3600/auth
Method POST
Request body 
{
  "email": "login@email.com",
  "password": "password"
}
Response body 
{
  "accessToken": "eyJhbG ... VlEZ5ew9I",
  "refreshToken": "VTBRKz ... M0E9PQ=="
}
Response status code 201

# Request body fields

email

The e-mail address that is the login for the user.

password

The user's password

# Response body fields

accessToken

The access token for the logged in user. This token will have to be passed in to every other API in order for the API to execute.

refreshToken

Not used for now. Future use will be to allow renewal of expired access tokens.

# Example usage

    ///////////////////////////////////////////////////////////////////////////////
// The following code invokes the Login API and extracts the access token    //
// from the API response. The token will be required for all other API calls //
///////////////////////////////////////////////////////////////////////////////
const axios = require('axios');

let accessToken = '';

axios.post(
  'http://localhost:3600/auth',
  {
    email: 'john.doe@org.com',
    password: 'password',
  },
)
  .then((res) => {
    console.log('Authentication successful');
    { accessToken } = res.data;
  })
  .catch(() => { 
    console.log('Authentication failed');
   });

    ###############################################################################
# The following code invokes the Login API and prints the access and refresh  #
# tokens received in the API response. The access token is required for all   #
# other API calls, while the refresh token may be used in the future to renew #
# an expired access token                                                     #
###############################################################################

# importing the requests library
import requests

# setup the data to be passed to the API
data = {'email': 'john.doe@org.com',
        'password': 'password'
}

# sending post request and saving response as response object
r = requests.post(url = 'http://localhost/3600/auth', data = data)

# get the returned data
ret = r.json()

# print the returned data
print(f"Access token: {ret.accessToken}")
print(f"Refresh token: {ret.refreshToken}")
