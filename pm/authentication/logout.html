# Logout

# API information
URL localhost:3600/logout
Method POST
Request body empty
Request headers Authorization: <access-token>
Response body 
Logout successful
Response status code 200

# Example usage

    ///////////////////////////////////////////////////////////////////////////////
// The following code invokes the Logout API to terminate a session and      //
// invalidate the current access token                                       //
///////////////////////////////////////////////////////////////////////////////
const axios = require('axios');

axios.post(
  'http://localhost:3600/auth',
  {
    email: 'john.doe@org.com',
    password: 'password',
  },
)
  .then((res) => {
    // Authentication successful
    const { accessToken } = res.body;

    // Logging out
    axios.post(
      'http://localhost:3600/auth',
      undefined,
      {
        headers: {
          Authorization: `Bearer ${accessToken}`,
        },
      },
    )
      .then((res) => console.log(res.data))
      .catch(() => console.log('Logout failed'));
  })
  .catch(() => { 
    console.log('Authentication failed');
   });

    ###############################################################################
# The following code invokes the Logout API to terminate a session and        #
# invalidate the current access token                                         #
###############################################################################

# importing the requests library
import requests

# setup the data to be passed to the API
data = {'email': 'john.doe@org.com',
        'password': 'password'
}

# sending login request and saving response as response object
r = requests.post(url = 'http://localhost/3600/auth', data = data)

# get the login response
loginResponse = r.json()

# sending the logout request and getting the response
headers  = { 'Authorization': f'Bearer {loginResponse.accessToken}' }
r = requests.post(url = 'http://localhost/3600/logout', headers = headers)

# print the status
if r.status_code == 200
  print('Logout successful')
else
  print('Logout failed')
